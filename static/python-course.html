    1. Python中 10 / 3 的结果是什么类型？为什么？
    float (3.333...)。/ 总是返回浮点数，确保除法精度不丢失。整除用 //。
    2. bool("") 和 bool("False") 分别返回什么？
    bool("") = False（空字符串是Falsy）；bool("False") = True（非空字符串都是Truthy）。

    from sklearn.datasets import load_iris from sklearn.model_selection import train_test_split from sklearn.ensemble import RandomForestClassifier from sklearn.metrics import classification_report iris = load_iris() X_train, X_test, y_train, y_test = train_test_split( iris.data, iris.target, test_size=0.2, random_state=42) rf = RandomForestClassifier(n_estimators=100).fit(X_train, y_train) y_pred = rf.predict(X_test) print(f"准确率: {rf.score(X_test, y_test)*100:.1f}%") print(classification_report(y_test, y_pred, target_names=iris.target_names)) # K-means聚类 from sklearn.cluster import KMeans km = KMeans(n_clusters=3, n_init=10).fit(iris.data) print(f"K-means聚类中心:\n{km.cluster_centers_}")
